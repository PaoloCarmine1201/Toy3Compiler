program
    def sommaIntera(a:int; b:int):int {
        return a + b;
    }

    def sommaDouble(a:double; b:double):double {
        return a + b;
    }

    def stampaMenuScelta() {
        "1) Addizione intera" !>>;
        "2) Addizione double" !>>;
        "0) End" !>>;
    }

    def inputIntero() :int {
        a:int;
        "Inserisci numero intero:" !>>;
        a <<;
        return a;
    }

    def inputDouble() :double {
        b:double;
        "Inserisci numero double:" !>>;
        b <<;
        return b;
    }

begin
    a | b | risultatointero: int;
    c | d | risultatodouble: double;
    scelta := 1;
    stampaMenuScelta();
    scelta <<;
    while(scelta <> 0) do {
        if(scelta == 1) then {
            a := inputIntero();
            b := inputIntero();
            risultatointero := sommaIntera(a, b);
            risultatointero !>>;
        }
        if(scelta == 2) then {
            c := inputDouble();
            d := inputDouble();
            risultatodouble := sommaDouble(c, d);
            risultatodouble !>>;
        }
        stampaMenuScelta();
        scelta << ;
    }
end
